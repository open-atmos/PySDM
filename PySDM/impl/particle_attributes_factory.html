 19    @staticmethod
 20    def attributes(particulator, req_attr, attributes):
 21        # pylint: disable=too-many-locals
 22        idx = particulator.Index.identity_index(particulator.n_sd)
 23
 24        extensive_attr = []
 25        maximum_attr = []
 26        for attr_name in req_attr:
 27            if isinstance(req_attr[attr_name], ExtensiveAttribute):
 28                extensive_attr.append(attr_name)
 29            elif isinstance(req_attr[attr_name], MaximumAttribute):
 30                maximum_attr.append(attr_name)
 31            elif not isinstance(
 32                req_attr[attr_name],
 33                (
 34                    DerivedAttribute,
 35                    get_attribute_class("multiplicity"),
 36                    CellAttribute,
 37                    DummyAttribute,
 38                ),
 39            ):
 40                raise AssertionError()
 41
 42        extensive_attribute_storage = particulator.IndexedStorage.empty(
 43            idx, (len(extensive_attr), particulator.n_sd), float
 44        )
 45        maximum_attributes = particulator.IndexedStorage.empty(
 46            idx, (len(maximum_attr), particulator.n_sd), float
 47        )
 48
 49        for attr in req_attr.values():
 50            if isinstance(attr, (DerivedAttribute, DummyAttribute)):
 51                if attr.name in attributes:
 52                    raise ValueError(
 53                        f"attribute '{attr.name}' is a dummy/derived one,"
 54                        f" but values were provided"
 55                    )
 56                attr.allocate(idx)
 57
 58        extensive_keys = {}
 59        maximum_keys = {}
 60
 61        def helper(req_attr, all_attr, names, data, keys):
 62            for i, attr in enumerate(names):
 63                keys[attr] = i
 64                req_attr[attr].set_data(data[i, :])
 65                try:
 66                    req_attr[attr].init(all_attr[attr])
 67                except KeyError as err:
 68                    raise ValueError(
 69                        f"attribute '{attr}' requested by one of the components"
 70                        f" but no initial values given"
 71                    ) from err
 72
 73        helper(
 74            req_attr,
 75            attributes,
 76            extensive_attr,
 77            extensive_attribute_storage,
 78            extensive_keys,
 79        )
 80        helper(req_attr, attributes, maximum_attr, maximum_attributes, maximum_keys)
 81
 82        n = req_attr["multiplicity"]
 83        n.allocate(idx)
 84        n.init(attributes["multiplicity"])
 85        req_attr["multiplicity"].data = particulator.IndexedStorage.indexed(idx, n.data)
 86        cell_id = req_attr["cell id"]
 87        cell_id.allocate(idx)
 88        cell_id.init(attributes["cell id"])
 89        req_attr["cell id"].data = particulator.IndexedStorage.indexed(
 90            idx, cell_id.data
 91        )
 92        try:
 93            cell_origin = req_attr["cell origin"]
 94            cell_origin.allocate(idx)
 95            cell_origin.init(attributes["cell origin"])
 96            req_attr["cell origin"].data = particulator.IndexedStorage.indexed(
 97                idx, cell_origin.data
 98            )
 99        except KeyError:
100            cell_origin = None
101        try:
102            position_in_cell = req_attr["position in cell"]
103            position_in_cell.allocate(idx)
104            position_in_cell.init(attributes["position in cell"])
105            req_attr["position in cell"].data = particulator.IndexedStorage.indexed(
106                idx, position_in_cell.data
107            )
108        except KeyError:
109            position_in_cell = None
110
111        cell_start = np.empty(particulator.mesh.n_cell + 1, dtype=int)
112
113        return ParticleAttributes(
114            particulator=particulator,
115            idx=idx,
116            extensive_attribute_storage=extensive_attribute_storage,
117            extensive_keys=extensive_keys,
118            # maximum_attributes, maximum_keys, # TODO #594
119            cell_start=cell_start,
120            attributes=req_attr,
121        )
122
123    @staticmethod
124    def empty_particles(particles, n_sd) -> ParticleAttributes:
125        idx = particles.Index.identity_index(n_sd)
126        return ParticleAttributes(
127            particulator=particles,
128            idx=idx,
129            extensive_attribute_storage=None,
130            extensive_keys={},
131            cell_start=np.zeros(2, dtype=np.int64),
132            attributes={},

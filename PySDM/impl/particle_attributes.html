 13class ParticleAttributes:  # pylint: disable=too-many-instance-attributes
 14    def __init__(
 15        self,
 16        *,
 17        particulator,
 18        idx,
 19        extensive_attribute_storage,
 20        extensive_keys: dict,
 21        cell_start,
 22        attributes: Dict[str, Attribute],
 23    ):
 24        self.__valid_n_sd = particulator.n_sd
 25        self.__healthy_memory = particulator.Storage.from_ndarray(np.full((1,), 1))
 26        self.__idx = idx
 27
 28        self.__extensive_attribute_storage = extensive_attribute_storage
 29        self.__extensive_keys = extensive_keys
 30
 31        self.cell_idx = particulator.Index.identity_index(len(cell_start) - 1)
 32        self.__cell_start = particulator.Storage.from_ndarray(cell_start)
 33        self.__cell_caretaker = particulator.backend.make_cell_caretaker(
 34            self.__idx.shape,
 35            self.__idx.dtype,
 36            len(self.__cell_start),
 37            scheme=particulator.sorting_scheme,
 38        )
 39        self.__sorted = False
 40        self.__attributes = attributes
 41
 42    @property
 43    def healthy(self) -> bool:
 44        return bool(self.__healthy_memory[0])
 45
 46    @healthy.setter
 47    def healthy(self, value: bool):
 48        self.__healthy_memory[:] = value
 49
 50    @property
 51    def cell_start(self):
 52        if not self.__sorted:
 53            self.__sort_by_cell_id()
 54        return self.__cell_start
 55
 56    @property
 57    def super_droplet_count(self):
 58        """returns the number of super-droplets in the system
 59        (which might differ from the initial one due to precipitation
 60        or removal during collision of multiplicity-of-one particles)"""
 61        assert self.healthy
 62        return len(self.__idx)
 63
 64    def mark_updated(self, key):
 65        self.__attributes[key].mark_updated()
 66
 67    def sanitize(self):
 68        if not self.healthy:
 69            self.__idx.length = self.__valid_n_sd
 70            self.__idx.remove_zero_n_or_flagged(self["multiplicity"])
 71            self.__valid_n_sd = self.__idx.length
 72            self.healthy = True
 73            self.__sorted = False
 74
 75    def cut_working_length(self, length):
 76        assert length <= len(self.__idx)
 77        self.__idx.length = length
 78
 79    def get_working_length(self):
 80        return len(self.__idx)
 81
 82    def reset_working_length(self):
 83        self.__idx.length = self.__valid_n_sd
 84
 85    def reset_cell_idx(self):
 86        self.cell_idx.reset_index()
 87        self.__sort_by_cell_id()
 88
 89    def keys(self):
 90        return self.__attributes.keys()
 91
 92    def __getitem__(self, item):
 93        return self.__attributes[item].get()
 94
 95    def __contains__(self, key):
 96        return key in self.__attributes
 97
 98    def permutation(self, u01, local):
 99        """apply Fisher-Yates algorithm to all super-droplets (local=False) or
100        otherwise on a per-cell basis"""
101        if local:
102            self.__idx.shuffle(u01, parts=self.cell_start)
103        else:
104            self.__idx.shuffle(u01)
105            self.__sorted = False
106
107    def __sort_by_cell_id(self):
108        self.__cell_caretaker(
109            self["cell id"], self.cell_idx, self.__cell_start, self.__idx
110        )
111        self.__sorted = True
112
113    def get_extensive_attribute_storage(self):
114        return self.__extensive_attribute_storage
115
116    def get_extensive_attribute_keys(self):
117        return self.__extensive_keys.keys()
118
119    def has_attribute(self, attr):
120        return attr in self.__attributes
121
122    def reset_idx(self):
123        self.__valid_n_sd = self.__idx.shape[0]
124        self.__idx.reset_index()
125        self.healthy = False

 18    def _deposition(self):
 19        assert self.formulae.particle_shape_and_density.supports_mixed_phase()
 20
 21        formulae = self.formulae_flattened
 22        multiplier = 2
 23        midpoint = True
 24        rel_tol_rh = 1e-2
 25        fuse = 16

 27        @numba.njit(**{**self.default_jit_flags, **{"parallel": False}})
 28        def calc_saturation_ratio_ice_temperature_and_pressure(
 29            vapour_mixing_ratio, dry_air_potential_temperature, dry_air_density
 30        ):
 31            temperature = formulae.state_variable_triplet__T(
 32                rhod=dry_air_density,
 33                thd=dry_air_potential_temperature,
 34            )
 35            total_pressure = formulae.state_variable_triplet__p(
 36                rhod=dry_air_density,
 37                T=temperature,
 38                water_vapour_mixing_ratio=vapour_mixing_ratio,
 39            )
 40            vapour_partial_pressure = formulae.state_variable_triplet__pv(
 41                p=total_pressure,
 42                water_vapour_mixing_ratio=vapour_mixing_ratio,
 43            )
 44            saturation_ratio_ice = (
 45                vapour_partial_pressure
 46                / formulae.saturation_vapour_pressure__pvs_ice(temperature)
 47            )
 48            return saturation_ratio_ice, temperature, total_pressure

 50        @numba.njit(**{**self.default_jit_flags, **{"parallel": False}})
 51        def mass_deposition_rate_per_droplet(
 52            temperature: float,
 53            rho_d: float,
 54            signed_mass_old: float,
 55            latent_heat_sub: float,
 56            saturation_ratio_ice: float,
 57            pressure: float,
 58        ):
 59            radius = formulae.particle_shape_and_density__mass_to_radius(
 60                signed_mass_old
 61            )
 62            pvs_ice = formulae.saturation_vapour_pressure__pvs_ice(temperature)
 63
 64            capacity = formulae.diffusion_ice_capacity__capacity(abs(signed_mass_old))
 65
 66            mass_ventilation_factor = 1  # TODO #1655
 67            heat_ventilation_factor = mass_ventilation_factor  # TODO #1588
 68
 69            Dv_const = formulae.diffusion_thermics__D(temperature, pressure)
 70            lambdaD = formulae.diffusion_ice_kinetics__lambdaD(temperature, pressure)
 71            diffusion_coefficient = formulae.diffusion_ice_kinetics__D(
 72                Dv_const, radius, lambdaD, temperature
 73            )
 74
 75            Ka_const = formulae.diffusion_thermics__K(temperature, pressure)
 76            lambdaK = formulae.diffusion_ice_kinetics__lambdaK(temperature, pressure)
 77            thermal_conductivity = formulae.diffusion_ice_kinetics__K(
 78                Ka_const, radius, lambdaK, temperature, rho_d
 79            )
 80
 81            Fk = formulae.drop_growth__Fk(
 82                T=temperature,
 83                K=thermal_conductivity * heat_ventilation_factor,
 84                lv=latent_heat_sub,
 85            )
 86            Fd = formulae.drop_growth__Fd(
 87                T=temperature,
 88                D=diffusion_coefficient * mass_ventilation_factor,
 89                pvs=pvs_ice,
 90            )
 91            howell_factor_x_diffcoef_x_rhovsice_x_icess = (
 92                formulae.drop_growth__r_dr_dt(
 93                    RH_eq=1,
 94                    RH=saturation_ratio_ice,
 95                    Fk=Fk,
 96                    Fd=Fd,
 97                )
 98                * formulae.constants.rho_w
 99            )
100
101            mass_deposition_rate = (
102                4 * np.pi * capacity * howell_factor_x_diffcoef_x_rhovsice_x_icess
103            )
104            if mass_deposition_rate > 1:
105                print(
106                    radius,
107                    Fk,
108                    Fd,
109                    mass_deposition_rate,
110                    howell_factor_x_diffcoef_x_rhovsice_x_icess,
111                    pvs_ice,
112                )
113                assert False
114            return mass_deposition_rate

116        @numba.njit(**{**self.default_jit_flags, **{"parallel": False}})
117        def _loop(
118            fake,
119            temperature,
120            rhod,
121            thd,
122            signed_water_mass,
123            saturation_ratio_ice,
124            total_pressure,
125            multiplicity,
126            sub_time_step,
127            mass_of_dry_air,
128        ):
129            latent_heat_sub = formulae.latent_heat_sublimation__ls(temperature)
130            delta_rv = 0
131            for i, ksi in enumerate(multiplicity):
132                if not formulae.trivia__unfrozen(signed_water_mass[i]):
133                    mass_deposition_rate = mass_deposition_rate_per_droplet(
134                        temperature=temperature,
135                        rho_d=rhod,
136                        signed_mass_old=signed_water_mass[i],
137                        latent_heat_sub=latent_heat_sub,
138                        saturation_ratio_ice=saturation_ratio_ice,
139                        pressure=total_pressure,
140                    )
141                    delta_rv += (
142                        -mass_deposition_rate * ksi * sub_time_step / mass_of_dry_air
143                    )
144                    if not fake:
145                        x_old = formulae.diffusion_coordinate__x(-signed_water_mass[i])
146                        dx_dt_old = formulae.diffusion_coordinate__dx_dt(
147                            -signed_water_mass[i], mass_deposition_rate
148                        )
149                        x_new = formulae.trivia__explicit_euler(
150                            x_old, sub_time_step, dx_dt_old
151                        )
152                        signed_water_mass[i] = -formulae.diffusion_coordinate__mass(
153                            x_new
154                        )
155                        if x_new > 1:
156                            print(x_old, dx_dt_old, x_new, signed_water_mass[i])
157                            assert False
158            delta_thd = (
159                formulae.state_variable_triplet__dthd_dt(
160                    rhod=rhod,
161                    thd=thd,
162                    T=temperature,
163                    d_water_vapour_mixing_ratio__dt=delta_rv / sub_time_step,
164                    lv=latent_heat_sub,
165                )
166                * sub_time_step
167            )
168            if delta_rv == 0:
169                assert delta_thd == 0
170            else:
171                assert (delta_rv < 0 < delta_thd) or (delta_rv > 0 > delta_thd)
172            return delta_rv, delta_thd

174        @numba.njit(**{**self.default_jit_flags, **{"parallel": False}})
175        def body(  # pylint: disable=too-many-arguments
176            *,
177            adaptive,
178            multiplicity,
179            signed_water_mass,
180            current_vapour_mixing_ratio,
181            current_dry_air_density,
182            current_dry_potential_temperature,
183            cell_volume,
184            time_step,
185            cell_id,
186            # to be modified
187            predicted_vapour_mixing_ratio,
188            predicted_dry_potential_temperature,
189            predicted_dry_air_density,
190        ):
191            """simplest adaptivity:
192            - global dt - no cell-wise logic (we don't have any test/example for it!)
193            - no mechanism to retain dt value over timesteps
194            - explicit Euler mass integration (vs. implicit in condensation)
195            """
196            # pylint: disable=too-many-locals
197            n_substeps = 1
198            cid = cell_id[0]  # TODO #1524: add support for multi-cell environments
199
200            rv_tendency = (
201                predicted_vapour_mixing_ratio[cid] - current_vapour_mixing_ratio[cid]
202            ) / time_step
203            thd_tendency = (
204                predicted_dry_potential_temperature[cid]
205                - current_dry_potential_temperature[cid]
206            ) / time_step
207            rhod_tendency = (
208                predicted_dry_air_density[cid] - current_dry_air_density[cid]
209            ) / time_step
210            dry_air_mass_mean = (
211                cell_volume
212                * (predicted_dry_air_density[cid] + current_dry_air_density[cid])
213                / 2
214            )
215
216            if adaptive:
217                n_substeps = 1 / multiplier
218                delta_rh_long = np.nan
219                for burnout in range(fuse + 1):
220                    if burnout == fuse:
221                        assert False
222                    sub_time_step = time_step / n_substeps
223                    rhod = (
224                        current_dry_air_density[cid]
225                        + rhod_tendency * (0.5 if midpoint else 1) * sub_time_step
226                    )
227                    rv = (
228                        current_vapour_mixing_ratio[cid]
229                        + rv_tendency * (0.5 if midpoint else 1) * sub_time_step
230                    )
231                    thd = (
232                        current_dry_potential_temperature[cid]
233                        + thd_tendency * (0.5 if midpoint else 1) * sub_time_step
234                    )
235
236                    saturation_ratio_ice, temperature, total_pressure = (
237                        calc_saturation_ratio_ice_temperature_and_pressure(
238                            vapour_mixing_ratio=rv,
239                            dry_air_potential_temperature=thd,
240                            dry_air_density=rhod,
241                        )
242                    )
243                    delta_rv, delta_thd = _loop(
244                        fake=True,
245                        temperature=temperature,
246                        rhod=rhod,
247                        thd=thd,
248                        signed_water_mass=signed_water_mass,
249                        saturation_ratio_ice=saturation_ratio_ice,
250                        total_pressure=total_pressure,
251                        multiplicity=multiplicity,
252                        sub_time_step=sub_time_step,
253                        mass_of_dry_air=dry_air_mass_mean,
254                    )
255                    delta_rh_short = (
256                        calc_saturation_ratio_ice_temperature_and_pressure(
257                            vapour_mixing_ratio=rv + delta_rv,
258                            dry_air_potential_temperature=thd + delta_thd,
259                            dry_air_density=rhod,
260                        )[0]
261                        - saturation_ratio_ice
262                    )
263                    if (
264                        n_substeps < 1
265                        or rv < -delta_rv
266                        or not formulae.trivia__within_tolerance(
267                            abs(delta_rh_long - multiplier * delta_rh_short),
268                            saturation_ratio_ice,
269                            rel_tol_rh,
270                        )
271                    ):
272                        delta_rh_long = delta_rh_short
273                        n_substeps *= multiplier
274                    else:
275                        break
276            sub_time_step = time_step / n_substeps
277
278            rv = current_vapour_mixing_ratio[cid]
279            thd = current_dry_potential_temperature[cid]
280            rhod = current_dry_air_density[cid]
281
282            assert n_substeps == int(n_substeps)
283            for _ in range(int(n_substeps)):
284                rv += sub_time_step * rv_tendency * (0.5 if midpoint else 1)
285                thd += sub_time_step * thd_tendency * (0.5 if midpoint else 1)
286                rhod += sub_time_step * rhod_tendency * (0.5 if midpoint else 1)
287
288                saturation_ratio_ice, temperature, total_pressure = (
289                    calc_saturation_ratio_ice_temperature_and_pressure(
290                        vapour_mixing_ratio=rv,
291                        dry_air_potential_temperature=thd,
292                        dry_air_density=rhod,
293                    )
294                )
295                delta_rv, delta_thd = _loop(
296                    fake=False,
297                    temperature=temperature,
298                    rhod=rhod,
299                    thd=thd,
300                    signed_water_mass=signed_water_mass,
301                    saturation_ratio_ice=saturation_ratio_ice,
302                    total_pressure=total_pressure,
303                    multiplicity=multiplicity,
304                    sub_time_step=sub_time_step,
305                    mass_of_dry_air=dry_air_mass_mean,
306                )
307                thd += delta_thd
308                rv += delta_rv
309                assert rv >= 0
310
311                if midpoint:
312                    thd += sub_time_step * thd_tendency / 2
313                    rv += sub_time_step * rv_tendency / 2
314                    rhod += sub_time_step * rhod_tendency / 2
315
316            predicted_dry_potential_temperature[cid] = thd
317            predicted_vapour_mixing_ratio[cid] = rv
318
319        return body

290                    calc_saturation_ratio_ice_temperature_and_pressure(
291                        vapour_mixing_ratio=rv,
292                        dry_air_potential_temperature=thd,
293                        dry_air_density=rhod,
294                    )
295                )
296                delta_rv, delta_thd = _loop(
297                    fake=False,
298                    temperature=temperature,
299                    rhod=rhod,
300                    thd=thd,
301                    signed_water_mass=signed_water_mass,
302                    saturation_ratio_ice=saturation_ratio_ice,
303                    total_pressure=total_pressure,
304                    multiplicity=multiplicity,
305                    sub_time_step=sub_time_step,
306                    mass_of_dry_air=dry_air_mass_mean,
307                )
308                thd += delta_thd
309                rv += delta_rv
310                assert rv >= 0
311
312                if midpoint:
313                    thd += sub_time_step * thd_tendency / 2
314                    rv += sub_time_step * rv_tendency / 2
315                    rhod += sub_time_step * rhod_tendency / 2
316
317            predicted_dry_potential_temperature[cid] = thd
318            predicted_vapour_mixing_ratio[cid] = rv
319
320        return body

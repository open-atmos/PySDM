 16@numba.njit(**{**conf.JIT_FLAGS, **{"parallel": False}})
 17def pair_indices(i, idx, is_first_in_pair, prob_like):
 18    """given permutation array `idx` and `is_first_in_pair` flag array,
 19    returns indices `j` and `k` of droplets within pair `i` and a `skip_pair` flag,
 20    `j` points to the droplet that is first in pair (higher or equal multiplicity)
 21    output is valid only if `2*i` or `2*i+1` points to a valid pair start index (within one cell)
 22    otherwise the `skip_pair` flag is set to True and returned `j` & `k` indices are set to -1.
 23    In addition, the `prob_like` array is checked for zeros at position `i`, in which case
 24    the `skip_pair` is also set to `True`
 25    """
 26    skip_pair = False
 27
 28    if prob_like[i] == 0:
 29        skip_pair = True
 30        j, k = -1, -1
 31    else:
 32        offset = 1 - is_first_in_pair[2 * i]
 33        j = idx[2 * i + offset]
 34        k = idx[2 * i + 1 + offset]
 35    return j, k, skip_pair

 44@numba.njit(**{**conf.JIT_FLAGS, **{"parallel": False}})
 45def coalesce(  # pylint: disable=too-many-arguments
 46    i, j, k, cid, multiplicity, gamma, attributes, coalescence_rate
 47):
 48    atomic_add(coalescence_rate, cid, gamma[i] * multiplicity[k])
 49    new_n = multiplicity[j] - gamma[i] * multiplicity[k]
 50    if new_n > 0:
 51        multiplicity[j] = new_n
 52        for a in range(len(attributes)):
 53            attributes[a, k] += gamma[i] * attributes[a, j]
 54    else:  # new_n == 0
 55        multiplicity[j] = multiplicity[k] // 2
 56        multiplicity[k] = multiplicity[k] - multiplicity[j]
 57        for a in range(len(attributes)):
 58            attributes[a, j] = gamma[i] * attributes[a, j] + attributes[a, k]
 59            attributes[a, k] = attributes[a, j]

 62@numba.njit(**{**conf.JIT_FLAGS, **{"parallel": False}})
 63def compute_transfer_multiplicities(
 64    gamma, j, k, multiplicity, particle_mass, fragment_mass_i, max_multiplicity
 65):  # pylint: disable=too-many-arguments
 66    overflow_flag = False
 67    gamma_j_k = 0
 68    take_from_j_test = multiplicity[k]
 69    take_from_j = 0
 70    new_mult_k_test = (
 71        (particle_mass[j] + particle_mass[k]) / fragment_mass_i
 72    ) * multiplicity[k]
 73    new_mult_k = multiplicity[k]
 74    for m in range(int(gamma)):
 75        # check for overflow of multiplicity
 76        if new_mult_k_test > max_multiplicity:
 77            overflow_flag = True
 78            break
 79
 80        # check for new_n >= 0
 81        if take_from_j_test > multiplicity[j]:
 82            break
 83
 84        take_from_j = take_from_j_test
 85        new_mult_k = new_mult_k_test
 86        gamma_j_k = m + 1
 87
 88        take_from_j_test += new_mult_k_test
 89        new_mult_k_test = (
 90            new_mult_k_test * (particle_mass[j] / fragment_mass_i) + new_mult_k_test
 91        )
 92
 93    return take_from_j, new_mult_k, gamma_j_k, overflow_flag

 96@numba.njit(**{**conf.JIT_FLAGS, **{"parallel": False}})
 97def get_new_multiplicities_and_update_attributes(
 98    j, k, attributes, multiplicity, take_from_j, new_mult_k
 99):  # pylint: disable=too-many-arguments
100    for a in range(len(attributes)):
101        attributes[a, k] *= multiplicity[k]
102        attributes[a, k] += take_from_j * attributes[a, j]
103        attributes[a, k] /= new_mult_k
104
105    if multiplicity[j] > take_from_j:
106        nj = multiplicity[j] - take_from_j
107        nk = new_mult_k
108
109    else:  # take_from_j == multiplicity[j]
110        nj = new_mult_k / 2
111        nk = nj
112        for a in range(len(attributes)):
113            attributes[a, j] = attributes[a, k]
114    return nj, nk

117@numba.njit(**{**conf.JIT_FLAGS, **{"parallel": False}})
118def round_multiplicities_to_ints_and_update_attributes(
119    j,
120    k,
121    nj,
122    nk,
123    attributes,
124    multiplicity,
125):  # pylint: disable=too-many-arguments
126    multiplicity[j] = max(round(nj), 1)
127    multiplicity[k] = max(round(nk), 1)
128    factor_j = nj / multiplicity[j]
129    factor_k = nk / multiplicity[k]
130    for a in range(len(attributes)):
131        attributes[a, k] *= factor_k
132        attributes[a, j] *= factor_j

135@numba.njit(**{**conf.JIT_FLAGS, **{"parallel": False}})
136def break_up(  # pylint: disable=too-many-arguments,c,too-many-locals
137    i,
138    j,
139    k,
140    cid,
141    multiplicity,
142    gamma,
143    attributes,
144    fragment_mass,
145    max_multiplicity,
146    breakup_rate,
147    breakup_rate_deficit,
148    warn_overflows,
149    particle_mass,
150):  # breakup0 guarantees take_from_j <= multiplicity[j]
151    take_from_j, new_mult_k, gamma_j_k, overflow_flag = compute_transfer_multiplicities(
152        gamma[i],
153        j,
154        k,
155        multiplicity,
156        particle_mass,
157        fragment_mass[i],
158        max_multiplicity,
159    )
160    gamma_deficit = gamma[i] - gamma_j_k
161
162    # breakup1 also handles new_n[j] == 0 case via splitting
163    nj, nk = get_new_multiplicities_and_update_attributes(
164        j, k, attributes, multiplicity, take_from_j, new_mult_k
165    )
166
167    atomic_add(breakup_rate, cid, gamma_j_k * multiplicity[k])
168    atomic_add(breakup_rate_deficit, cid, gamma_deficit * multiplicity[k])
169
170    # breakup2 also guarantees that no multiplicities are set to 0
171    round_multiplicities_to_ints_and_update_attributes(
172        j, k, nj, nk, attributes, multiplicity
173    )
174    if overflow_flag and warn_overflows:
175        warn("overflow", __file__)

178@numba.njit(**{**conf.JIT_FLAGS, **{"parallel": False}})
179def break_up_while(
180    i,
181    j,
182    k,
183    cid,
184    multiplicity,
185    gamma,
186    attributes,
187    fragment_mass,
188    max_multiplicity,
189    breakup_rate,
190    breakup_rate_deficit,
191    warn_overflows,
192    particle_mass,
193):  # pylint: disable=too-many-arguments,unused-argument,too-many-locals
194    gamma_deficit = gamma[i]
195    overflow_flag = False
196    while gamma_deficit > 0:
197        if multiplicity[k] == multiplicity[j]:
198            take_from_j = multiplicity[j]
199            new_mult_k = (
200                (particle_mass[j] + particle_mass[k])
201                / fragment_mass[i]
202                * multiplicity[k]
203            )
204
205            # check for overflow
206            if new_mult_k > max_multiplicity:
207                atomic_add(breakup_rate_deficit, cid, gamma_deficit * multiplicity[k])
208                overflow_flag = True
209                break
210            gamma_j_k = gamma_deficit
211
212        else:
213            if multiplicity[k] > multiplicity[j]:
214                j, k = k, j
215            (
216                take_from_j,
217                new_mult_k,
218                gamma_j_k,
219                overflow_flag,
220            ) = compute_transfer_multiplicities(
221                gamma_deficit,
222                j,
223                k,
224                multiplicity,
225                particle_mass,
226                fragment_mass[i],
227                max_multiplicity,
228            )
229
230        nj, nk = get_new_multiplicities_and_update_attributes(
231            j, k, attributes, multiplicity, take_from_j, new_mult_k
232        )
233
234        atomic_add(breakup_rate, cid, gamma_j_k * multiplicity[k])
235        gamma_deficit -= gamma_j_k
236        round_multiplicities_to_ints_and_update_attributes(
237            j, k, nj, nk, attributes, multiplicity
238        )
239
240    atomic_add(breakup_rate_deficit, cid, gamma_deficit * multiplicity[k])
241
242    if overflow_flag and warn_overflows:
243        warn("overflow", __file__)

331    def _scale_prob_for_adaptive_sdm_gamma_body(self):
332        @numba.njit(**self.default_jit_flags)
333        # pylint: disable=too-many-arguments,too-many-locals
334        def body(
335            prob,
336            idx,
337            length,
338            multiplicity,
339            cell_id,
340            dt_left,
341            dt,
342            dt_range,
343            is_first_in_pair,
344            stats_n_substep,
345            stats_dt_min,
346        ):
347            dt_todo = np.empty_like(dt_left)
348            for cid in numba.prange(len(dt_todo)):  # pylint: disable=not-an-iterable
349                dt_todo[cid] = min(dt_left[cid], dt_range[1])
350            for i in range(length // 2):  # TODO #571
351                j, k, skip_pair = pair_indices(i, idx, is_first_in_pair, prob)
352                if skip_pair:
353                    continue
354                prop = multiplicity[j] // multiplicity[k]
355                dt_optimal = dt * prop / prob[i]
356                cid = cell_id[j]
357                dt_optimal = max(dt_optimal, dt_range[0])
358                dt_todo[cid] = min(dt_todo[cid], dt_optimal)
359                stats_dt_min[cid] = min(stats_dt_min[cid], dt_optimal)
360            for i in numba.prange(length // 2):  # pylint: disable=not-an-iterable
361                j, _, skip_pair = pair_indices(i, idx, is_first_in_pair, prob)
362                if skip_pair:
363                    continue
364                prob[i] *= dt_todo[cell_id[j]] / dt
365            for cid in numba.prange(len(dt_todo)):  # pylint: disable=not-an-iterable
366                dt_left[cid] -= dt_todo[cid]

514    @cached_property
515    def _compute_gamma_body(self):
516        @numba.njit(**self.default_jit_flags)
517        # pylint: disable=too-many-arguments,too-many-locals
518        def body(
519            prob,
520            rand,
521            idx,
522            length,
523            multiplicity,
524            cell_id,
525            collision_rate_deficit,
526            collision_rate,
527            is_first_in_pair,
528            out,
529        ):
530            """
531            return in "out" array gamma (see: http://doi.org/10.1002/qj.441, section 5)
532            formula:
533            gamma = floor(prob) + 1 if rand <  prob - floor(prob)
534                  = floor(prob)     if rand >= prob - floor(prob)
535
536            out may point to the same array as prob
537            """
538            for i in numba.prange(length // 2):  # pylint: disable=not-an-iterable
539                out[i] = np.ceil(prob[i] - rand[i])
540                j, k, skip_pair = pair_indices(i, idx, is_first_in_pair, out)
541                if skip_pair:
542                    continue
543                prop = multiplicity[j] // multiplicity[k]
544                g = min(int(out[i]), prop)
545                cid = cell_id[j]
546                atomic_add(collision_rate, cid, g * multiplicity[k])
547                atomic_add(
548                    collision_rate_deficit, cid, (int(out[i]) - g) * multiplicity[k]
549                )
550                out[i] = g
551
552        return body

626    def _normalize_body(self):
627        @numba.njit(**{**self.default_jit_flags, **{"parallel": False}})
628        # pylint: disable=too-many-arguments
629        def body(prob, cell_id, cell_idx, cell_start, norm_factor, timestep, dv):
630            n_cell = cell_start.shape[0] - 1
631            for i in range(n_cell):
632                sd_num = cell_start[i + 1] - cell_start[i]
633                if sd_num < 2:
634                    norm_factor[i] = 0
635                else:
636                    norm_factor[i] = (
637                        timestep / dv * sd_num * (sd_num - 1) / 2 / (sd_num // 2)
638                    )
639            for d in numba.prange(prob.shape[0]):  # pylint: disable=not-an-iterable
640                prob[d] *= norm_factor[cell_idx[cell_id[d]]]
641
642        return body

656    @cached_property
657    def remove_zero_n_or_flagged(self):
658        @numba.njit(**{**self.default_jit_flags, **{"parallel": False}})
659        def body(multiplicity, idx, length) -> int:
660            flag = len(idx)
661            new_length = length
662            i = 0
663            while i < new_length:
664                if idx[i] == flag or multiplicity[idx[i]] == 0:
665                    new_length -= 1
666                    idx[i] = idx[new_length]
667                    idx[new_length] = flag
668                else:
669                    i += 1

532            return in "out" array gamma (see: http://doi.org/10.1002/qj.441, section 5)
533            formula:
534            gamma = floor(prob) + 1 if rand <  prob - floor(prob)
535                  = floor(prob)     if rand >= prob - floor(prob)
536
537            out may point to the same array as prob
538            """
539            for i in numba.prange(length // 2):  # pylint: disable=not-an-iterable
540                out[i] = np.ceil(prob[i] - rand[i])
541                j, k, skip_pair = pair_indices(i, idx, is_first_in_pair, out)
542                if skip_pair:
543                    continue
544                prop = multiplicity[j] // multiplicity[k]
545                g = min(int(out[i]), prop)
546                cid = cell_id[j]
547                atomic_add(collision_rate, cid, g * multiplicity[k])
548                atomic_add(
549                    collision_rate_deficit, cid, (int(out[i]) - g) * multiplicity[k]
550                )
551                out[i] = g
552
553        return body

631            n_cell = cell_start.shape[0] - 1

693    @numba.njit(**conf.JIT_FLAGS)
694    # pylint: disable=too-many-arguments
695    def _parallel_counting_sort_by_cell_id_and_update_cell_start(
696        new_idx, idx, cell_id, cell_idx, length, cell_start, cell_start_p
697    ):
698        cell_end_thread = cell_start_p
699        # Warning: Assuming len(cell_end) == n_cell+1
700        thread_num = cell_end_thread.shape[0]
701        for t in numba.prange(thread_num):  # pylint: disable=not-an-iterable
702            cell_end_thread[t, :] = 0
703            for i in range(
704                t * length // thread_num,
705                (t + 1) * length // thread_num if t < thread_num - 1 else length,
706            ):
707                cell_end_thread[t, cell_idx[cell_id[idx[i]]]] += 1
708
709        cell_start[:] = np.sum(cell_end_thread, axis=0)
710        for i in range(1, len(cell_start)):
711            cell_start[i] += cell_start[i - 1]
712
713        tmp = cell_end_thread[0, :]
714        tmp[:] = cell_end_thread[thread_num - 1, :]
715        cell_end_thread[thread_num - 1, :] = cell_start[:]
716        for t in range(thread_num - 2, -1, -1):
717            cell_start[:] = cell_end_thread[t + 1, :] - tmp[:]
718            tmp[:] = cell_end_thread[t, :]
719            cell_end_thread[t, :] = cell_start[:]
720
721        for t in numba.prange(thread_num):  # pylint: disable=not-an-iterable
722            for i in range(
723                (
724                    (t + 1) * length // thread_num - 1
725                    if t < thread_num - 1
726                    else length - 1
727                ),
728                t * length // thread_num - 1,
729                -1,
730            ):
731                cell_end_thread[t, cell_idx[cell_id[idx[i]]]] -= 1
732                new_idx[cell_end_thread[t, cell_idx[cell_id[idx[i]]]]] = idx[i]

737    def _linear_collection_efficiency_body(self):
738        @numba.njit(**self.default_jit_flags)
739        # pylint: disable=too-many-arguments,too-many-locals
740        def body(params, output, radii, is_first_in_pair, idx, length, unit):
741            A, B, D1, D2, E1, E2, F1, F2, G1, G2, G3, Mf, Mg = params
742            output[:] = 0
743            for i in numba.prange(length - 1):  # pylint: disable=not-an-iterable
744                if is_first_in_pair[i]:
745                    if radii[idx[i]] > radii[idx[i + 1]]:
746                        r = radii[idx[i]] / unit
747                        r_s = radii[idx[i + 1]] / unit
748                    else:
749                        r = radii[idx[i + 1]] / unit
750                        r_s = radii[idx[i]] / unit
751                    p = r_s / r
752                    if p not in (0, 1):
753                        G = (G1 / r) ** Mg + G2 + G3 * r
754                        Gp = (1 - p) ** G
755                        if Gp != 0:
756                            D = D1 / r**D2
757                            E = E1 / r**E2
758                            F = (F1 / r) ** Mf + F2
759                            output[i // 2] = A + B * p + D / p**F + E / Gp
760                            output[i // 2] = max(0, output[i // 2])
761
762        return body

 12@numba.njit(**{**conf.JIT_FLAGS, **{"parallel": False}})
 13def straub_Nr(  # pylint: disable=too-many-arguments,unused-argument
 14    i,
 15    Nr1,
 16    Nr2,
 17    Nr3,
 18    Nr4,
 19    Nrt,
 20    CW,
 21    gam,
 22):  # pylint: disable=too-many-branches`
 23    if gam[i] * CW[i] >= 7.0:
 24        Nr1[i] = 0.088 * (gam[i] * CW[i] - 7.0)
 25    if CW[i] >= 21.0:
 26        Nr2[i] = 0.22 * (CW[i] - 21.0)
 27        if CW[i] <= 46.0:
 28            Nr3[i] = 0.04 * (46.0 - CW[i])
 29    else:
 30        Nr3[i] = 1.0
 31    Nr4[i] = 1.0
 32    Nrt[i] = Nr1[i] + Nr2[i] + Nr3[i] + Nr4[i]

 35@numba.njit(**{**conf.JIT_FLAGS, **{"parallel": False}})
 36def straub_mass_remainder(  # pylint: disable=too-many-arguments,unused-argument
 37    i, vl, ds, mu1, sigma1, mu2, sigma2, mu3, sigma3, d34, Nr1, Nr2, Nr3, Nr4
 38):
 39    # pylint: disable=too-many-arguments, too-many-locals
 40    Nr1[i] = Nr1[i] * np.exp(3 * mu1 + 9 * np.power(sigma1, 2) / 2)
 41    Nr2[i] = Nr2[i] * (mu2**3 + 3 * mu2 * sigma2**2)
 42    Nr3[i] = Nr3[i] * (mu3**3 + 3 * mu3 * sigma3**2)
 43    Nr4[i] = vl[i] * 6 / np.pi + ds[i] ** 3 - Nr1[i] - Nr2[i] - Nr3[i]
 44    if Nr4[i] <= 0.0:
 45        d34[i] = 0
 46        Nr4[i] = 0
 47    else:
 48        d34[i] = np.exp(np.log(Nr4[i]) / 3)

 77    def _fragmentation_limiters_body(self):
 78        @numba.njit(**self.default_jit_flags)
 79        # pylint: disable=too-many-arguments
 80        def body(n_fragment, frag_volume, vmin, nfmax, x_plus_y):
 81            for i in numba.prange(len(frag_volume)):  # pylint: disable=not-an-iterable
 82                if x_plus_y[i] == 0.0:
 83                    frag_volume[i] = 0.0
 84                    n_fragment[i] = 1.0
 85                else:
 86                    if np.isnan(frag_volume[i]) or frag_volume[i] == 0.0:
 87                        frag_volume[i] = x_plus_y[i]
 88                    frag_volume[i] = min(frag_volume[i], x_plus_y[i])
 89                    if nfmax is not None and x_plus_y[i] / frag_volume[i] > nfmax:
 90                        frag_volume[i] = x_plus_y[i] / nfmax
 91                    elif frag_volume[i] < vmin:
 92                        frag_volume[i] = x_plus_y[i]
 93                    n_fragment[i] = x_plus_y[i] / frag_volume[i]
 94
 95        return body

322    def _straub_fragmentation_body(self):
323        ff = self.formulae_flattened
324
325        @numba.njit(**self.default_jit_flags)
326        def body(
327            *, CW, gam, ds, v_max, frag_volume, rand, Nr1, Nr2, Nr3, Nr4, Nrt, d34
328        ):  # pylint: disable=too-many-arguments,too-many-locals
329            for i in numba.prange(len(frag_volume)):  # pylint: disable=not-an-iterable
330                straub_Nr(i, Nr1, Nr2, Nr3, Nr4, Nrt, CW, gam)
331                sigma1 = ff.fragmentation_function__params_sigma1(CW[i])
332                mu1 = ff.fragmentation_function__params_mu1(sigma1)
333                sigma2 = ff.fragmentation_function__params_sigma2(CW[i])
334                mu2 = ff.fragmentation_function__params_mu2(ds[i])
335                sigma3 = ff.fragmentation_function__params_sigma3(CW[i])
336                mu3 = ff.fragmentation_function__params_mu3(ds[i])
337                straub_mass_remainder(
338                    i,
339                    v_max,
340                    ds,
341                    mu1,
342                    sigma1,
343                    mu2,
344                    sigma2,
345                    mu3,
346                    sigma3,
347                    d34,
348                    Nr1,
349                    Nr2,
350                    Nr3,
351                    Nr4,
352                )
353                Nrt[i] = Nr1[i] + Nr2[i] + Nr3[i] + Nr4[i]
354
355                if Nrt[i] == 0.0:
356                    diameter = 0.0
357                else:
358                    if rand[i] < Nr1[i] / Nrt[i]:
359                        X = rand[i] * Nrt[i] / Nr1[i]
360                        lnarg = mu1 + np.sqrt(2) * sigma1 * ff.trivia__erfinv_approx(X)
361                        diameter = np.exp(lnarg)
362                    elif rand[i] < (Nr2[i] + Nr1[i]) / Nrt[i]:
363                        X = (rand[i] * Nrt[i] - Nr1[i]) / Nr2[i]
364                        diameter = mu2 + np.sqrt(2) * sigma2 * ff.trivia__erfinv_approx(
365                            X
366                        )
367                    elif rand[i] < (Nr3[i] + Nr2[i] + Nr1[i]) / Nrt[i]:
368                        X = (rand[i] * Nrt[i] - Nr1[i] - Nr2[i]) / Nr3[i]
369                        diameter = mu3 + np.sqrt(2) * sigma3 * ff.trivia__erfinv_approx(
370                            X
371                        )
372                    else:
373                        diameter = d34[i]
374
375                frag_volume[i] = diameter**3 * ff.constants.PI / 6
376
377        return body

379    @cached_property
380    def _ll82_fragmentation_body(self):  # pylint: disable=too-many-statements
381        ff = self.formulae_flattened
382
383        @numba.njit(**self.default_jit_flags)
384        def body(
385            *, CKE, W, W2, St, ds, dl, dcoal, frag_volume, rand, Rf, Rs, Rd, tol
386        ):  # pylint: disable=too-many-branches,too-many-locals,too-many-statements
387            for i in numba.prange(len(frag_volume)):  # pylint: disable=not-an-iterable
388                if dl[i] <= 0.4e-3:
389                    frag_volume[i] = dcoal[i] ** 3 * ff.constants.PI / 6
390                elif ds[i] == 0.0 or dl[i] == 0.0:
391                    frag_volume[i] = 1e-18
392                else:
393                    ll82_Nr(i, Rf, Rs, Rd, CKE, W, W2)
394                    if rand[i] <= Rf[i]:  # filament breakup
395                        (H1, mu1, sigma1) = ff.fragmentation_function__params_f1(
396                            dl[i], dcoal[i]
397                        )
398                        (H2, mu2, sigma2) = ff.fragmentation_function__params_f2(ds[i])
399                        (H3, mu3, sigma3) = ff.fragmentation_function__params_f3(
400                            ds[i], dl[i]
401                        )
402                        H1 = H1 * mu1
403                        H2 = H2 * mu2
404                        H3 = H3 * np.exp(mu3)
405                        Hsum = H1 + H2 + H3
406                        rand[i] = rand[i] / Rf[i]
407                        if rand[i] <= H1 / Hsum:
408                            X = max(rand[i] * Hsum / H1, tol)
409                            frag_volume[i] = mu1 + np.sqrt(
410                                2
411                            ) * sigma1 * ff.trivia__erfinv_approx(2 * X - 1)
412                        elif rand[i] <= (H1 + H2) / Hsum:
413                            X = (rand[i] * Hsum - H1) / H2
414                            frag_volume[i] = mu2 + np.sqrt(
415                                2
416                            ) * sigma2 * ff.trivia__erfinv_approx(2 * X - 1)
417                        else:
418                            X = min((rand[i] * Hsum - H1 - H2) / H3, 1.0 - tol)
419                            lnarg = mu3 + np.sqrt(
420                                2
421                            ) * sigma3 * ff.trivia__erfinv_approx(2 * X - 1)
422                            frag_volume[i] = np.exp(lnarg)
423
424                    elif rand[i] <= Rf[i] + Rs[i]:  # sheet breakup
425                        (H1, mu1, sigma1) = ff.fragmentation_function__params_s1(
426                            dl[i], ds[i], dcoal[i]
427                        )
428                        (H2, mu2, sigma2) = ff.fragmentation_function__params_s2(
429                            dl[i], ds[i], St[i]
430                        )
431                        H1 = H1 * mu1
432                        H2 = H2 * np.exp(mu2)
433                        Hsum = H1 + H2
434                        rand[i] = (rand[i] - Rf[i]) / (Rs[i])
435                        if rand[i] <= H1 / Hsum:
436                            X = max(rand[i] * Hsum / H1, tol)
437                            frag_volume[i] = mu1 + np.sqrt(
438                                2
439                            ) * sigma1 * ff.trivia__erfinv_approx(2 * X - 1)
440                        else:
441                            X = min((rand[i] * Hsum - H1) / H2, 1.0 - tol)
442                            lnarg = mu2 + np.sqrt(
443                                2
444                            ) * sigma2 * ff.trivia__erfinv_approx(2 * X - 1)
445                            frag_volume[i] = np.exp(lnarg)
446
447                    else:  # disk breakup
448                        (H1, mu1, sigma1) = ff.fragmentation_function__params_d1(
449                            W[i], dl[i], dcoal[i], CKE[i]
450                        )
451                        (H2, mu2, sigma2) = ff.fragmentation_function__params_d2(
452                            ds[i], dl[i], CKE[i]
453                        )
454                        H1 = H1 * mu1
455                        Hsum = H1 + H2
456                        rand[i] = (rand[i] - Rf[i] - Rs[i]) / Rd[i]
457                        if rand[i] <= H1 / Hsum:
458                            X = max(rand[i] * Hsum / H1, tol)
459                            frag_volume[i] = mu1 + np.sqrt(
460                                2
461                            ) * sigma1 * ff.trivia__erfinv_approx(2 * X - 1)
462                        else:
463                            X = min((rand[i] * Hsum - H1) / H2, 1 - tol)
464                            lnarg = mu2 + np.sqrt(
465                                2
466                            ) * sigma2 * ff.trivia__erfinv_approx(2 * X - 1)
467                            frag_volume[i] = np.exp(lnarg)
468
469                    frag_volume[i] = (
470                        frag_volume[i] * 0.01
471                    )  # diameter in cm; convert to m
472                    frag_volume[i] = frag_volume[i] ** 3 * ff.constants.PI / 6
473
474        return body

322    @cached_property
323    def _straub_fragmentation_body(self):
324        ff = self.formulae_flattened
325
326        @numba.njit(**self.default_jit_flags)
327        def body(
328            *, CW, gam, ds, v_max, frag_volume, rand, Nr1, Nr2, Nr3, Nr4, Nrt, d34
329        ):  # pylint: disable=too-many-arguments,too-many-locals
330            for i in numba.prange(len(frag_volume)):  # pylint: disable=not-an-iterable
331                straub_Nr(i, Nr1, Nr2, Nr3, Nr4, Nrt, CW, gam)
332                sigma1 = ff.fragmentation_function__params_sigma1(CW[i])
333                mu1 = ff.fragmentation_function__params_mu1(sigma1)
334                sigma2 = ff.fragmentation_function__params_sigma2(CW[i])
335                mu2 = ff.fragmentation_function__params_mu2(ds[i])
336                sigma3 = ff.fragmentation_function__params_sigma3(CW[i])
337                mu3 = ff.fragmentation_function__params_mu3(ds[i])
338                straub_mass_remainder(
339                    i,
340                    v_max,
341                    ds,
342                    mu1,
343                    sigma1,
344                    mu2,
345                    sigma2,
346                    mu3,
347                    sigma3,
348                    d34,
349                    Nr1,
350                    Nr2,
351                    Nr3,
352                    Nr4,
353                )
354                Nrt[i] = Nr1[i] + Nr2[i] + Nr3[i] + Nr4[i]
355
356                if Nrt[i] == 0.0:
357                    diameter = 0.0
358                else:
359                    if rand[i] < Nr1[i] / Nrt[i]:
360                        X = rand[i] * Nrt[i] / Nr1[i]
361                        lnarg = mu1 + np.sqrt(2) * sigma1 * ff.trivia__erfinv_approx(X)
362                        diameter = np.exp(lnarg)
363                    elif rand[i] < (Nr2[i] + Nr1[i]) / Nrt[i]:
364                        X = (rand[i] * Nrt[i] - Nr1[i]) / Nr2[i]
365                        diameter = mu2 + np.sqrt(2) * sigma2 * ff.trivia__erfinv_approx(
366                            X
367                        )
368                    elif rand[i] < (Nr3[i] + Nr2[i] + Nr1[i]) / Nrt[i]:
369                        X = (rand[i] * Nrt[i] - Nr1[i] - Nr2[i]) / Nr3[i]
370                        diameter = mu3 + np.sqrt(2) * sigma3 * ff.trivia__erfinv_approx(
371                            X
372                        )
373                    else:
374                        diameter = d34[i]
375
376                frag_volume[i] = diameter**3 * ff.constants.PI / 6
377
378        return body

380    @cached_property
381    def _ll82_fragmentation_body(self):  # pylint: disable=too-many-statements
382        ff = self.formulae_flattened
383
384        @numba.njit(**self.default_jit_flags)
385        def body(
386            *, CKE, W, W2, St, ds, dl, dcoal, frag_volume, rand, Rf, Rs, Rd, tol
387        ):  # pylint: disable=too-many-branches,too-many-locals,too-many-statements
388            for i in numba.prange(len(frag_volume)):  # pylint: disable=not-an-iterable
389                if dl[i] <= 0.4e-3:
390                    frag_volume[i] = dcoal[i] ** 3 * ff.constants.PI / 6
391                elif ds[i] == 0.0 or dl[i] == 0.0:
392                    frag_volume[i] = 1e-18
393                else:
394                    ll82_Nr(i, Rf, Rs, Rd, CKE, W, W2)
395                    if rand[i] <= Rf[i]:  # filament breakup
396                        (H1, mu1, sigma1) = ff.fragmentation_function__params_f1(
397                            dl[i], dcoal[i]
398                        )
399                        (H2, mu2, sigma2) = ff.fragmentation_function__params_f2(ds[i])
400                        (H3, mu3, sigma3) = ff.fragmentation_function__params_f3(
401                            ds[i], dl[i]
402                        )
403                        H1 = H1 * mu1
404                        H2 = H2 * mu2
405                        H3 = H3 * np.exp(mu3)
406                        Hsum = H1 + H2 + H3
407                        rand[i] = rand[i] / Rf[i]
408                        if rand[i] <= H1 / Hsum:
409                            X = max(rand[i] * Hsum / H1, tol)
410                            frag_volume[i] = mu1 + np.sqrt(
411                                2
412                            ) * sigma1 * ff.trivia__erfinv_approx(2 * X - 1)
413                        elif rand[i] <= (H1 + H2) / Hsum:
414                            X = (rand[i] * Hsum - H1) / H2
415                            frag_volume[i] = mu2 + np.sqrt(
416                                2
417                            ) * sigma2 * ff.trivia__erfinv_approx(2 * X - 1)
418                        else:
419                            X = min((rand[i] * Hsum - H1 - H2) / H3, 1.0 - tol)
420                            lnarg = mu3 + np.sqrt(
421                                2
422                            ) * sigma3 * ff.trivia__erfinv_approx(2 * X - 1)
423                            frag_volume[i] = np.exp(lnarg)
424
425                    elif rand[i] <= Rf[i] + Rs[i]:  # sheet breakup
426                        (H1, mu1, sigma1) = ff.fragmentation_function__params_s1(
427                            dl[i], ds[i], dcoal[i]
428                        )
429                        (H2, mu2, sigma2) = ff.fragmentation_function__params_s2(
430                            dl[i], ds[i], St[i]
431                        )
432                        H1 = H1 * mu1
433                        H2 = H2 * np.exp(mu2)
434                        Hsum = H1 + H2
435                        rand[i] = (rand[i] - Rf[i]) / (Rs[i])
436                        if rand[i] <= H1 / Hsum:
437                            X = max(rand[i] * Hsum / H1, tol)
438                            frag_volume[i] = mu1 + np.sqrt(
439                                2
440                            ) * sigma1 * ff.trivia__erfinv_approx(2 * X - 1)
441                        else:
442                            X = min((rand[i] * Hsum - H1) / H2, 1.0 - tol)
443                            lnarg = mu2 + np.sqrt(
444                                2
445                            ) * sigma2 * ff.trivia__erfinv_approx(2 * X - 1)
446                            frag_volume[i] = np.exp(lnarg)
447
448                    else:  # disk breakup
449                        (H1, mu1, sigma1) = ff.fragmentation_function__params_d1(
450                            W[i], dl[i], dcoal[i], CKE[i]
451                        )
452                        (H2, mu2, sigma2) = ff.fragmentation_function__params_d2(
453                            ds[i], dl[i], CKE[i]
454                        )
455                        H1 = H1 * mu1
456                        Hsum = H1 + H2
457                        rand[i] = (rand[i] - Rf[i] - Rs[i]) / Rd[i]
458                        if rand[i] <= H1 / Hsum:
459                            X = max(rand[i] * Hsum / H1, tol)
460                            frag_volume[i] = mu1 + np.sqrt(
461                                2
462                            ) * sigma1 * ff.trivia__erfinv_approx(2 * X - 1)
463                        else:
464                            X = min((rand[i] * Hsum - H1) / H2, 1 - tol)
465                            lnarg = mu2 + np.sqrt(
466                                2
467                            ) * sigma2 * ff.trivia__erfinv_approx(2 * X - 1)
468                            frag_volume[i] = np.exp(lnarg)
469
470                    frag_volume[i] = (
471                        frag_volume[i] * 0.01
472                    )  # diameter in cm; convert to m
473                    frag_volume[i] = frag_volume[i] ** 3 * ff.constants.PI / 6
474
475        return body

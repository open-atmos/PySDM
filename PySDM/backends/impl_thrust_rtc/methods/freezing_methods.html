 15    @cached_property
 16    def immersion_freezing_time_dependent_body(self):
 17        return trtc.For(
 18            param_names=(
 19                "rand",
 20                "immersed_surface_area",
 21                "signed_water_mass",
 22                "timestep",
 23                "cell",
 24                "a_w_ice",
 25                "relative_humidity",
 26            ),
 27            name_iter="i",
 28            body=f"""
 29                if (immersed_surface_area[i] == 0) {{
 30                    return;
 31                }}
 32                if ({self.formulae.trivia.unfrozen_and_saturated.c_inline(
 33                        signed_water_mass="signed_water_mass[i]",
 34                        relative_humidity="relative_humidity[cell[i]]"
 35                    )}) {{
 36                    auto rate_assuming_constant_temperature_within_dt = {self.formulae.heterogeneous_ice_nucleation_rate.j_het.c_inline(
 37                        a_w_ice="a_w_ice[cell[i]]"
 38                    )} * immersed_surface_area[i];
 39                    auto prob = 1 - {self.formulae.trivia.poissonian_avoidance_function.c_inline(
 40                        r="rate_assuming_constant_temperature_within_dt",
 41                        dt="timestep"
 42                    )};
 43                    if (rand[i] < prob) {{
 44                        signed_water_mass[i] = -1 * signed_water_mass[i];
 45                    }}
 46                }}
 47            """.replace(
 48                "real_type", self._get_c_type()
 49            ),
 50        )

143                    )}) {{
144                    auto da_w_ice = (relative_humidity_ice[cell[i]] - 1.0) * a_w_ice[cell[i]];
145                    if  ({self.formulae.homogeneous_ice_nucleation_rate.d_a_w_ice_within_range.c_inline(
146                        da_w_ice="da_w_ice"
147                    )}) {{
148                        auto da_w_ice = {self.formulae.homogeneous_ice_nucleation_rate.d_a_w_ice_maximum.c_inline(
149                            da_w_ice="da_w_ice"

151                        auto rate_assuming_constant_temperature_within_dt = {self.formulae.homogeneous_ice_nucleation_rate.j_hom.c_inline(
152                            T="temperature[cell[i]]",
153                            da_w_ice="da_w_ice"
154                        )} * volume[i];
155                        auto prob = 1 - {self.formulae.trivia.poissonian_avoidance_function.c_inline(
156                            r="rate_assuming_constant_temperature_within_dt",
157                            dt="timestep"
158                        )};
159                        if (rand[i] < prob) {{
160                            signed_water_mass[i] = -1 * signed_water_mass[i];
161                        }}
162                    }}

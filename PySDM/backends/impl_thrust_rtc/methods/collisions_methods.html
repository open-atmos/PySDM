  16struct Commons {
  17  static __device__ void flag_zero_multiplicity(
  18    int64_t j,
  19    int64_t k,
  20    VectorView<int64_t> multiplicity,
  21    VectorView<int64_t> healthy
  22  ) {
  23    if (multiplicity[k] == 0 || multiplicity[j] == 0) {
  24        healthy[0] = 0;
  25    }
  26  }

  28  static __device__ void coalesce(
  29    int64_t i,
  30    int64_t j,
  31    int64_t k,
  32    VectorView<int64_t> cell_id,
  33    VectorView<int64_t> multiplicity,
  34    VectorView<real_type> gamma,
  35    VectorView<real_type> attributes,
  36    VectorView<int64_t> coalescence_rate,
  37    int64_t n_attr,
  38    int64_t n_sd
  39  ) {
  40    auto cid = cell_id[j];
  41    atomicAdd((unsigned long long int*)&coalescence_rate[cid], (unsigned long long int)(gamma[i] * multiplicity[k]));
  42    auto new_n = multiplicity[j] - gamma[i] * multiplicity[k];
  43    if (new_n > 0) {
  44        multiplicity[j] = new_n;
  45        for (auto attr = 0; attr < n_attr * n_sd; attr+=n_sd) {
  46            attributes[attr + k] += gamma[i] * attributes[attr + j];
  47        }
  48    }
  49    else {  // new_n == 0
  50        multiplicity[j] = (int64_t)(multiplicity[k] / 2);
  51        multiplicity[k] = multiplicity[k] - multiplicity[j];
  52        for (auto attr = 0; attr < n_attr * n_sd; attr+=n_sd) {
  53            attributes[attr + j] = gamma[i] * attributes[attr + j] + attributes[attr + k];
  54            attributes[attr + k] = attributes[attr + j];
  55        }
  56    }
  57  }
  58
  59  static __device__ auto pair_indices(
  60      int64_t i,
  61      int64_t *j,
  62      int64_t *k,
  63      VectorView<int64_t> idx,
  64      VectorView<bool> is_first_in_pair,
  65      VectorView<real_type> prob_like
  66  ) {
  67      if (prob_like[i] == 0) {
  68          return true;
  69      }
  70      auto offset = 1 - is_first_in_pair[2 * i];
  71      j[0] = idx[2 * i + offset];
  72      k[0] = idx[2 * i + 1 + offset];
  73      return false;
  74  }

  76  static __device__ auto compute_transfer_multiplicities(
  77    real_type gamma,
  78    int64_t j,
  79    int64_t k,
  80    VectorView<int64_t> multiplicity,
  81    VectorView<real_type> particle_mass,
  82    real_type fragment_mass_i,
  83    int64_t max_multiplicity,
  84
  85    real_type *take_from_j,

  88    real_type gamma_j_k = 0;
  89    real_type take_from_j_test = multiplicity[k];
  90    take_from_j[0] = 0;
  91    real_type new_mult_k_test = ((particle_mass[j] + particle_mass[k]) / fragment_mass_i * multiplicity[k]);
  92    new_mult_k[0] = multiplicity[k];
  93
  94    for (auto m = 0; m < (int64_t) (gamma); m += 1) {
  95        if (new_mult_k_test > max_multiplicity) {
  96            break;
  97        }
  98
  99        if (take_from_j_test > multiplicity[j]) {
 100            break;
 101        }
 102
 103        take_from_j[0] = take_from_j_test;
 104        new_mult_k[0] = new_mult_k_test;
 105        gamma_j_k = m + 1;
 106
 107        take_from_j_test += new_mult_k_test;
 108        new_mult_k_test = new_mult_k_test * (particle_mass[j] / fragment_mass_i) + new_mult_k_test;
 109    }
 110    return gamma_j_k;
 111  }

 114    int64_t j,
 115    int64_t k,

 140        }

 145    int64_t j,

 147    real_type nj,
 148    real_type nk,

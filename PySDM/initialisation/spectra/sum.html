14class Sum:
15    def __init__(self, spectra: tuple, interpolation_grid=None):
16        interpolation_grid = interpolation_grid or default_interpolation_grid
17        self.spectra = spectra
18        self.norm_factor = sum((s.norm_factor for s in self.spectra))
19        percentiles = [s.percentiles(interpolation_grid) for s in self.spectra]
20        cdf_arg = np.zeros(len(interpolation_grid) * len(self.spectra) + 1)
21        cdf_arg[1:] = np.concatenate(percentiles)
22        cdf = self.cumulative(cdf_arg) / self.norm_factor
23        self.inverse_cdf = interp1d(cdf, cdf_arg)

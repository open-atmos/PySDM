15@register_product()
16class CloudWaterContent(MomentProduct):
17    def __init__(
18        self, unit="kg/m^3", name=None, specific=False, liquid=True, ice=True
19    ):  # pylint: disable=too-many-arguments
20        super().__init__(unit=unit, name=name)
21        self.specific = specific
22        self.liquid = liquid
23        self.ice = ice
24
25    def _impl(self, **kwargs):
26        cwc = 0.0
27        if self.liquid:
28            self._download_moment_to_buffer(
29                attr="water mass", rank=1, filter_range=(0, np.inf)
30            )
31            mass = self.buffer.copy()
32
33            self._download_moment_to_buffer(
34                attr="water mass", rank=0, filter_range=(0, np.inf)
35            )
36            number = self.buffer
37            cwc += mass * number / self.particulator.mesh.dv
38
39        if self.ice:
40            self._download_moment_to_buffer(
41                attr="water mass",
42                rank=1,
43                filter_range=(-np.inf, 0),
44                filter_attr="signed water mass",
45            )
46            mass = self.buffer.copy()
47
48            self._download_moment_to_buffer(
49                attr="water mass",
50                rank=0,
51                filter_range=(-np.inf, 0),
52                filter_attr="signed water mass",
53            )
54            number = self.buffer
55            cwc += mass * number / self.particulator.mesh.dv
56
57        if self.specific:
58            self._download_to_buffer(self.particulator.environment["rhod"])
59            cwc /= self.buffer
60        return cwc

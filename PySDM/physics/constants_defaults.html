367def compute_derived_values(c: dict):
368    """
369    computes derived quantities such as molar mass ratios, etc.
370
371    water molar mass is computed from molecular masses and VSMOW isotope abundances
372    (and neglecting molecular binding energies)
373    for discussion, see:
374    - caption of Table 2.1 in [Gat 2010](https://doi.org/10.1142/p027)
375    - [IAPWS Guidelines](http://www.iapws.org/relguide/fundam.pdf)
376    """
377
378    c["Mv"] = (
379        (
380            1
381            - 2 * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_2H"])
382            - 2 * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_3H"])
383            - 1 * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_17O"])
384            - 1 * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_18O"])
385        )
386        * (c["M_1H"] * 2 + c["M_16O"])
387        + 2
388        * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_2H"])
389        * (c["M_2H"] + c["M_1H"] + c["M_16O"])
390        + 2
391        * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_3H"])
392        * (c["M_3H"] + c["M_1H"] + c["M_16O"])
393        + 1
394        * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_17O"])
395        * (c["M_1H"] * 2 + c["M_17O"])
396        + 1
397        * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_18O"])
398        * (c["M_1H"] * 2 + c["M_18O"])
399    )
400
401    c["eps"] = c["Mv"] / c["Md"]
402    c["Rd"] = c["R_str"] / c["Md"]
403    c["Rv"] = c["R_str"] / c["Mv"]
404
405    c["Rd_over_c_pd"] = c["Rd"] / c["c_pd"]
406
407    c["water_molar_volume"] = c["Mv"] / c["rho_w"]
408    c["rho_STP"] = c["p_STP"] / c["Rd"] / c["T_STP"]
409    c["H_u"] = c["M"] / c["p_STP"]

675def compute_derived_values(c: dict):
676    """
677    computes derived quantities such as molar mass ratios, etc.
678
679    water molar mass is computed from molecular masses and VSMOW isotope abundances
680    (and neglecting molecular binding energies)
681    for discussion, see:
682    - caption of Table 2.1 in [Gat 2010](https://doi.org/10.1142/p027)
683    - [IAPWS Guidelines](http://www.iapws.org/relguide/fundam.pdf)
684    """
685
686    c["M_1H2_16O"] = c["M_1H"] * 2 + c["M_16O"]
687    c["M_2H_1H_16O"] = c["M_2H"] + c["M_1H"] + c["M_16O"]
688    c["M_3H_1H_16O"] = c["M_3H"] + c["M_1H"] + c["M_16O"]
689    c["M_1H2_17O"] = c["M_1H"] * 2 + c["M_17O"]
690    c["M_1H2_18O"] = c["M_1H"] * 2 + c["M_18O"]
691
692    c["Mv"] = (
693        (
694            1
695            - 2 * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_2H"])
696            - 2 * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_3H"])
697            - 1 * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_17O"])
698            - 1 * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_18O"])
699        )
700        * c["M_1H2_16O"]
701        + 2
702        * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_2H"])
703        * c["M_2H_1H_16O"]
704        + 2
705        * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_3H"])
706        * c["M_3H_1H_16O"]
707        + 1 * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_17O"]) * c["M_1H2_17O"]
708        + 1 * Trivia.mixing_ratio_to_specific_content(c["VSMOW_R_18O"]) * c["M_1H2_18O"]
709    )
710
711    c["eps"] = c["Mv"] / c["Md"]
712    c["Rd"] = c["R_str"] / c["Md"]
713    c["Rv"] = c["R_str"] / c["Mv"]
714
715    c["Rd_over_c_pd"] = c["Rd"] / c["c_pd"]
716
717    c["water_molar_volume"] = c["Mv"] / c["rho_w"]
718    c["rho_STP"] = c["p_STP"] / c["Rd"] / c["T_STP"]
719    c["H_u"] = c["M"] / c["p_STP"]
720
721    c["l_tri"] = c["L_tri"] / c["Mv"]

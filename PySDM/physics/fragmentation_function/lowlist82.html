 39    def params_f3(const, ds, dl):  # pylint: disable=too-many-locals
 40        dsCM = ds / const.CM
 41        dlCM = dl / const.CM
 42        # eq (3.3), (3.4)
 43        Ff1 = max(
 44            0,
 45            (
 46                (-2.25e4 * (dlCM - 0.403) ** 2 - 37.9) * (dsCM) ** (2.5)
 47                + 9.67 * (dlCM - 0.170) ** 2
 48                + 4.95
 49            ),
 50        )
 51        Ff2 = 1.02e4 * dsCM ** (2.83) + 2
 52        # eq (3.5)
 53        ds0 = max(0.04, (Ff1 / 2.83) ** (1 / 1.02e4))
 54        if dsCM > ds0:
 55            Ff = max(2.0, Ff1)
 56        else:
 57            Ff = max(2.0, Ff2)
 58        Dff3 = 0.241 * (dsCM) + 0.0129  # (4.14)
 59        # eq (4.18) - (4.21)
 60        Pf301 = 1.68e5 * dsCM ** (2.33)
 61        Pf302 = max(
 62            0,
 63            (
 64                (43.4 * (dlCM + 1.81) ** 2 - 159.0) / dsCM
 65                - 3870 * (dlCM - 0.285) ** 2
 66                - 58.1
 67            ),
 68        )
 69        alpha = (dsCM - ds0) / (0.2 * ds0)
 70        Pf303 = alpha * Pf301 + (1 - alpha) * Pf302
 71        if dsCM < ds0:
 72            Pf0 = Pf301
 73        elif dsCM > 1.2 * ds0:
 74            Pf0 = Pf302
 75        else:
 76            Pf0 = Pf303
 77        # eq (4.22), (4.16), (4.17) (4.23)
 78        sigmaf3 = 10 * Dff3
 79        muf3 = np.log(Dff3) + sigmaf3**2
 80        Hf3 = Pf0 * Dff3 / np.exp(-0.5 * sigmaf3**2)
 81        for _ in range(10):
 82            if sigmaf3 == 0.0 or Hf3 == 0:
 83                return (0.0, np.log(ds0), np.log(ds0))
 84            sigmaf3 = (
 85                np.sqrt(2 / np.pi)
 86                * (Ff - 2)
 87                / Hf3
 88                / (1 - math.erf((np.log(0.01) - muf3) / np.sqrt(2) / sigmaf3))
 89            )
 90            muf3 = np.log(Dff3) + sigmaf3**2
 91            Hf3 = Pf0 * Dff3 / np.exp(-0.5 * sigmaf3**2)
 92
 93        return (Hf3, muf3, sigmaf3)

113    def params_s2(const, dl, ds, St):
114        dsCM = ds / const.CM
115        dlCM = dl / const.CM
116        Dss2 = (
117            0.254 * (dsCM ** (0.413)) * np.exp(3.53 * dsCM ** (2.51) * (dlCM - dsCM))
118        )  # (4.27)
119        bstar = 14.2 * np.exp(-17.2 * dsCM)
120        Ps20 = 0.23 * dsCM ** (-3.93) * dlCM ** (bstar)  # (4.29)
121        sigmas2 = 10 * Dss2  # as in (4.22)
122        mus2 = np.log(Dss2) + sigmas2**2  # (4.32)
123        Hs2 = Ps20 * Dss2 / np.exp(-0.5 * sigmas2**2)  # (4.28)
124
125        Fs = 5 * math.erf((St - 2.52e-6) / (1.85e-6)) + 6  # (3.7)
126
127        for _ in range(10):
128            sigmas2 = (
129                np.sqrt(2 / np.pi)
130                * (Fs - 1)
131                / Hs2
132                / (1 - math.erf((np.log(0.01) - mus2) / np.sqrt(2) / sigmas2))
133            )
134            mus2 = np.log(Dss2) + sigmas2**2  # (4.32)
135            Hs2 = Ps20 * Dss2 / np.exp(-0.5 * sigmas2**2)  # (4.28)
136
137        return (Hs2, mus2, sigmas2)

177                return (0.0, np.log(Ddd2), np.log(Ddd2))
178            sigmad2 = (
179                np.sqrt(2 / np.pi)
180                * (Fd - 1)
181                / Hd2
182                / (1 - math.erf((np.log(0.01) - mud2) / np.sqrt(2) / sigmad2))
183            )
184            mud2 = np.log(Ddd2) + sigmad2**2
185            Hd2 = Pd20 * Ddd2 / np.exp(-0.5 * sigmad2**2)
186
187        return (Hd2, mud2, sigmad2)

 90    @staticmethod
 91    def erfinv_approx(const, c):
 92        return (
 93            2
 94            * np.sqrt(const.VEDDER_1987_A)
 95            * np.sinh(
 96                np.arcsinh(
 97                    np.arctanh(c)
 98                    / 2
 99                    / const.VEDDER_1987_b
100                    / np.power(const.VEDDER_1987_A, const.ONE_AND_A_HALF)
101                )
102                / 3
103            )
104        )

108        return (delta + 1) * reference_ratio
109
110    @staticmethod
111    def isotopic_ratio_2_delta(ratio, reference_ratio):
112        return ratio / reference_ratio - 1

114    @staticmethod
115    def isotopic_enrichment_to_delta_SMOW(E, delta_0_SMOW):
116        """(see also eq. 10 in Pierchala et al. 2022)
117
118        conversion from E to delta_R_SMOW with:
119          _R/SMOW = R / R_SMOW - 1
120          E = _R/R0 = R / R0 - 1
121        and the sought formula (the quantity used to define d-excess, etc.) is:
122          _R/SMOW(E) = (E + 1) * R_0 / R_SMOW - 1
123                      = (E + 1) * (_R0/SMOW + 1) - 1
124        where:
125          _R0/SMOW is the initial SMOW-delta in the experiment
126        """
127        return (E + 1) * (delta_0_SMOW + 1) - 1
128
129    @staticmethod
130    def mixing_ratio_to_specific_content(mixing_ratio):
131        return mixing_ratio / (1 + mixing_ratio)
132
133    @staticmethod
134    def dn_dlogr(r, dn_dr):
135        return np.log(10) * r * dn_dr
136
137    @staticmethod
138    def air_schmidt_number(dynamic_viscosity, diffusivity, density):
139        return dynamic_viscosity / diffusivity / density
140
141    @staticmethod
142    def sqrt_re_times_cbrt_sc(const, Re, Sc):
143        return np.power(Re, const.ONE_HALF) * np.power(Sc, const.ONE_THIRD)
144
145    @staticmethod
146    def K2C(const, TK):
147        return TK - const.T0
148
149    @staticmethod
150    def C2K(const, TC):
151        return TC + const.T0
152
153    @staticmethod
154    def poissonian_avoidance_function(r, dt):
155        """cumulative probability of zero events occurring within time `dt`
156        (or void probability, or avoidance function) in a Poisson counting
157        process with a constant rate `r`
158        """
159        return np.exp(-r * dt)

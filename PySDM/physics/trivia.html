 50    @staticmethod
 51    def vant_hoff(const, K, dH, T, *, T_0):
 52        return K * np.exp(-dH / const.R_str * (1 / T - 1 / T_0))

 54    @staticmethod
 55    def tdep2enthalpy(const, tdep):
 56        return -tdep * const.R_str

 64        return specific_gravity * mole_fraction / (1 - mole_fraction)

 68        return mixing_ratio / (specific_gravity + mixing_ratio)

 94    @staticmethod
 95    def erfinv_approx(const, c):
 96        return (
 97            2
 98            * np.sqrt(const.VEDDER_1987_A)
 99            * np.sinh(
100                np.arcsinh(
101                    np.arctanh(c)
102                    / 2
103                    / const.VEDDER_1987_b
104                    / np.power(const.VEDDER_1987_A, const.ONE_AND_A_HALF)
105                )
106                / 3
107            )
108        )

110    @staticmethod
111    def isotopic_delta_2_ratio(delta, reference_ratio):
112        return (delta + 1) * reference_ratio
113
114    @staticmethod
115    def isotopic_ratio_2_delta(ratio, reference_ratio):
116        return ratio / reference_ratio - 1

118    @staticmethod
119    def isotopic_enrichment_to_delta_SMOW(E, delta_0_SMOW):
120        """(see also eq. 10 in Pierchala et al. 2022)
121
122        conversion from E to delta_R_SMOW with:
123          _R/SMOW = R / R_SMOW - 1
124          E = _R/R0 = R / R0 - 1
125        and the sought formula (the quantity used to define d-excess, etc.) is:
126          _R/SMOW(E) = (E + 1) * R_0 / R_SMOW - 1
127                      = (E + 1) * (_R0/SMOW + 1) - 1
128        where:
129          _R0/SMOW is the initial SMOW-delta in the experiment
130        """
131        return (E + 1) * (delta_0_SMOW + 1) - 1
132
133    @staticmethod
134    def mixing_ratio_to_specific_content(mixing_ratio):
135        return mixing_ratio / (1 + mixing_ratio)
136
137    @staticmethod
138    def dn_dlogr(r, dn_dr):
139        return np.log(10) * r * dn_dr
140
141    @staticmethod
142    def air_schmidt_number(dynamic_viscosity, diffusivity, density):
143        return dynamic_viscosity / diffusivity / density
144
145    @staticmethod
146    def sqrt_re_times_cbrt_sc(const, Re, Sc):
147        return np.power(Re, const.ONE_HALF) * np.power(Sc, const.ONE_THIRD)
148
149    @staticmethod
150    def K2C(const, TK):
151        return TK - const.T0
152
153    @staticmethod
154    def C2K(const, TC):
155        return TC + const.T0
156
157    @staticmethod
158    def poissonian_avoidance_function(r, dt):
159        """cumulative probability of zero events occurring within time `dt`
160        (or void probability, or avoidance function) in a Poisson counting
161        process with a constant rate `r`
162        """
163        return np.exp(-r * dt)
164
165    @staticmethod
166    def tau(Bo, dm_dt_over_m):
167        """
168        see text above Table 1 [Bolin 1958](https://digitallibrary.un.org/record/3892725)
169        """
170        return 1 / Bo / dm_dt_over_m

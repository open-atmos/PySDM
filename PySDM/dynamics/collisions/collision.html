PySDM.dynamics.collisions.collision

General algorithm format:

  1. Determine whether collision occurs
  2. If collision occurs: a. Determine whether coalescence, breakup, or bouncing occur Ec = coalescence efficiency Eb = collisional-breakup efficiency 1 - Ec - Eb = bounce back to original fragments (subset of breakup) b. Perform the relevant dynamic
  1"""
  2General algorithm format:
  31. Determine whether collision occurs
  42. If collision occurs:
  5    a. Determine whether coalescence, breakup, or bouncing occur
  6        Ec = coalescence efficiency
  7        Eb = collisional-breakup efficiency
  8        1 - Ec - Eb = bounce back to original fragments (subset of breakup)
  9    b. Perform the relevant dynamic
 10"""

174    def __call__(self):
175        if self.enable:
176            if not self.adaptive:
177                for _ in range(self.__substeps):
178                    self.step()
179            else:
180                self.dt_left[:] = self.particulator.dt
181
182                while self.particulator.attributes.get_working_length() != 0:
183                    self.particulator.attributes.cell_idx.sort_by_key(self.dt_left)
184                    self.step()
185                    self.particulator.attributes.cut_working_length(
186                        self.particulator.adaptive_sdm_end(self.dt_left)
187                    )
188
189                self.particulator.attributes.reset_working_length()
190                self.particulator.attributes.reset_cell_idx()
191            self.rnd_opt_coll.reset()
192            if self.enable_breakup:
193                self.rnd_opt_proc.reset()
194                self.rnd_opt_frag.reset()

196    def step(self):
197        pairs_rand, rand = self.rnd_opt_coll.get_random_arrays()
198
199        self.toss_candidate_pairs_and_sort_within_pair_by_multiplicity(
200            self.is_first_in_pair, pairs_rand
201        )
202
203        prob = self.gamma
204        self.compute_probabilities_of_collision(self.is_first_in_pair, out=prob)
205
206        if self.enable_breakup:
207            proc_rand = self.rnd_opt_proc.get_random_arrays()
208            rand_frag = self.rnd_opt_frag.get_random_arrays()
209            self.compute_coalescence_efficiency(self.Ec_temp, self.is_first_in_pair)
210            self.compute_breakup_efficiency(self.Eb_temp, self.is_first_in_pair)
211            self.compute_number_of_fragments(
212                self.n_fragment, self.fragment_mass, rand_frag, self.is_first_in_pair
213            )
214        else:
215            proc_rand = None
216
217        self.compute_gamma(
218            prob=prob, rand=rand, is_first_in_pair=self.is_first_in_pair, out=self.gamma
219        )
220
221        self.particulator.collision_coalescence_breakup(
222            enable_breakup=self.enable_breakup,
223            gamma=self.gamma,
224            rand=proc_rand,
225            Ec=self.Ec_temp,
226            Eb=self.Eb_temp,
227            fragment_mass=self.fragment_mass,
228            coalescence_rate=self.coalescence_rate,
229            breakup_rate=self.breakup_rate,
230            breakup_rate_deficit=self.breakup_rate_deficit,
231            is_first_in_pair=self.is_first_in_pair,
232            warn_overflows=self.warn_overflows,
233            max_multiplicity=self.max_multiplicity,
234        )

236    def toss_candidate_pairs_and_sort_within_pair_by_multiplicity(
237        self, is_first_in_pair, u01
238    ):
239        self.particulator.attributes.permutation(u01, self.croupier == "local")
240        is_first_in_pair.update(
241            self.particulator.attributes.cell_start,
242            self.particulator.attributes.cell_idx,
243            self.particulator.attributes["cell id"],
244        )
245        self.particulator.sort_within_pair_by_attr(
246            is_first_in_pair, attr_name="multiplicity"
247        )
248
249    def compute_probabilities_of_collision(self, is_first_in_pair, out):
250        """eq. (20) in [Shima et al. 2009](https://doi.org/10.1002/qj.441)"""
251        self.collision_kernel(self.kernel_temp, is_first_in_pair)
252        out.max(self.particulator.attributes["multiplicity"], is_first_in_pair)
253        out *= self.kernel_temp
254        self.particulator.normalize(out, self.norm_factor_temp)
255
256    def compute_n_fragment(self, n_fragment, u01, is_first_in_pair):
257        self.compute_number_of_fragments(n_fragment, u01, is_first_in_pair)
258
259    def compute_gamma(self, prob, rand, is_first_in_pair, out):
260        """see sec. 5.1.3 point (3) in [Shima et al. 2009](https://doi.org/10.1002/qj.441)
261        note that in PySDM gamma also serves the purpose of disabling collisions
262        for droplets without a pair (i.e. odd number of particles within a grid cell)
263        """
264        if self.adaptive:
265            self.particulator.backend.scale_prob_for_adaptive_sdm_gamma(
266                prob=prob,
267                multiplicity=self.particulator.attributes["multiplicity"],
268                cell_id=self.particulator.attributes["cell id"],
269                dt_left=self.dt_left,
270                dt=self.particulator.dt,
271                dt_range=self.dt_coal_range,
272                is_first_in_pair=is_first_in_pair,
273                stats_n_substep=self.stats_n_substep,
274                stats_dt_min=self.stats_dt_min,
275            )
276            if self.stats_dt_min.amin() == self.dt_coal_range[0]:
277                warnings.warn("adaptive time-step reached dt_min")
278        else:
279            prob /= self.__substeps
280
281        self.particulator.backend.compute_gamma(
282            prob=prob,
283            rand=rand,
284            multiplicity=self.particulator.attributes["multiplicity"],
285            cell_id=self.particulator.attributes["cell id"],
286            collision_rate_deficit=self.collision_rate_deficit,
287            collision_rate=self.collision_rate,
288            is_first_in_pair=is_first_in_pair,
289            out=out,
290        )

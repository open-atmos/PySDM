 12class NetCDFExporter_1d:  # pylint: disable=too-few-public-methods,too-many-instance-attributes
 13    def __init__(  # pylint: disable = too-many-arguments
 14        self, data, settings, simulator, filename, exclude_particle_reservoir=True
 15    ):
 16        self.data = data
 17        self.settings = settings
 18        self.simulator = simulator
 19        self.vars = None
 20        self.filename = filename
 21        self.nz_export = (
 22            int(self.settings.z_max / self.settings.dz)
 23            if exclude_particle_reservoir
 24            else settings.nz
 25        )
 26        self.z0 = (
 27            0.0 if exclude_particle_reservoir else -settings.particle_reservoir_depth
 28        )
 29        self.n_save_spec = len(self.settings.save_spec_and_attr_times)
 30
 31    def _write_settings(self, ncdf):
 32        for setting in dir(self.settings):
 33            setattr(ncdf, setting, getattr(self.settings, setting))
 34
 35    def _create_dimensions(self, ncdf):
 36        ncdf.createDimension("time", self.settings.nt + 1)
 37        ncdf.createDimension("height", self.nz_export)
 38
 39        if self.n_save_spec != 0:
 40            ncdf.createDimension("time_save_spec", self.n_save_spec)
 41            for name, instance in self.simulator.particulator.products.items():
 42                if len(instance.shape) == 2:
 43                    dim_name = name.replace(" ", "_") + "_bin_index"
 44                    ncdf.createDimension(dim_name, self.settings.number_of_bins)
 45
 46    def _create_variables(self, ncdf):
 47        self.vars = {}
 48        self.vars["time"] = ncdf.createVariable("time", "f", ["time"])
 49        self.vars["time"][:] = self.settings.dt * np.arange(self.settings.nt + 1)
 50        self.vars["time"].units = "seconds"
 51
 52        self.vars["height"] = ncdf.createVariable("height", "f", ["height"])
 53        self.vars["height"][:] = self.z0 + self.settings.dz * (
 54            1 / 2 + np.arange(self.nz_export)
 55        )
 56        self.vars["height"].units = "metres"
 57
 58        if self.n_save_spec != 0:
 59            self.vars["time_save_spec"] = ncdf.createVariable(
 60                "time_save_spec", "f", ["time_save_spec"]
 61            )
 62            self.vars["time_save_spec"][:] = self.settings.save_spec_and_attr_times
 63            self.vars["time_save_spec"].units = "seconds"
 64
 65            for name, instance in self.simulator.particulator.products.items():
 66                if len(instance.shape) == 2:
 67                    label = name.replace(" ", "_") + "_bin_index"
 68                    self.vars[label] = ncdf.createVariable(label, "f", (label,))
 69                    self.vars[label][:] = np.arange(1, self.settings.number_of_bins + 1)
 70
 71        for name, instance in self.simulator.particulator.products.items():
 72            if name in self.vars:
 73                raise AssertionError(
 74                    f"product ({name}) has same name as one of netCDF dimensions"
 75                )
 76
 77            n_dimensions = len(instance.shape)
 78            if n_dimensions == 0:
 79                dimensions = ("time",)
 80            elif n_dimensions == 1:
 81                dimensions = ("height", "time")
 82            elif n_dimensions == 2:
 83                dim_name = name.replace(" ", "_") + "_bin_index"
 84                if self.n_save_spec == 0:
 85                    continue
 86                if self.n_save_spec == 1:
 87                    dimensions = ("height", f"{dim_name}")
 88                else:
 89                    dimensions = ("height", f"{dim_name}", "time_save_spec")
 90            else:
 91                raise NotImplementedError()
 92
 93            self.vars[name] = ncdf.createVariable(name, "f", dimensions)
 94            self.vars[name].units = instance.unit
 95
 96    def _write_variables(self):
 97        for var in self.simulator.particulator.products.keys():
 98            n_dimensions = len(self.simulator.particulator.products[var].shape)
 99            if n_dimensions == 0:
100                self.vars[var][:] = self.data[var][:]
101            elif n_dimensions == 1:
102                self.vars[var][:, :] = self.data[var][-self.nz_export :, :]
103            elif n_dimensions == 2:
104                if self.n_save_spec == 0:
105                    continue
106                if self.n_save_spec == 1:
107                    self.vars[var][:, :] = self.data[var][-self.nz_export :, :, 0]
108                else:
109                    self.vars[var][:, :, :] = self.data[var][-self.nz_export :, :, :]
110            else:
111                raise NotImplementedError()
112
113    def run(self):
114        with netcdf_file(self.filename, mode="w") as ncdf:
115            self._write_settings(ncdf)
116            self._create_dimensions(ncdf)
117            self._create_variables(ncdf)
118            self._write_variables()

121def readNetCDF_1d(file):
122    f = netcdf_file(file, "r")
123    output_settings = deepcopy(f._attributes)
124
125    _products = deepcopy(f.variables)
126    output_products = {}
127    for k, v in _products.items():
128        output_products[k] = v[:]
129
130    output_dicts = namedtuple("output_dicts", "settings products")
131    outputs = output_dicts(output_settings, output_products)
132    f.close()
133    return outputs

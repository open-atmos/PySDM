 14class VTKExporter:
 15    """
 16    Example of use:
 17
 18    exporter = VTKExporter()
 19
 20    for step in range(settings.n_steps):
 21        simulation.particulator.run(1)
 22
 23        exporter.export_attributes(simulation.particulator)
 24        exporter.export_products(simulation.particulator)
 25
 26    """
 27
 28    def __init__(
 29        self,
 30        *,
 31        path=".",
 32        attributes_filename="sd_attributes",
 33        products_filename="sd_products",
 34        file_num_len=10,
 35        verbose=False,
 36    ):
 37        self.path = os.path.join(path, "output")
 38
 39        if not os.path.isdir(self.path):
 40            os.mkdir(self.path)
 41
 42        self.attributes_file_path = os.path.join(self.path, attributes_filename)
 43        self.products_file_path = os.path.join(self.path, products_filename)
 44        self.num_len = file_num_len
 45        self.exported_times = {}
 46        self.exported_times["attributes"] = {}
 47        self.exported_times["products"] = {}
 48        self.verbose = verbose
 49
 50    def write_pvd(self):
 51        pvd_attributes = VtkGroup(self.attributes_file_path)
 52        for k, v in self.exported_times["attributes"].items():
 53            pvd_attributes.addFile(k + ".vtu", sim_time=v)
 54        pvd_attributes.save()
 55
 56        pvd_products = VtkGroup(self.products_file_path)
 57        for k, v in self.exported_times["products"].items():
 58            pvd_products.addFile(k + ".vts", sim_time=v)
 59        pvd_products.save()
 60
 61    def export_attributes(self, particulator):
 62        path = (
 63            self.attributes_file_path
 64            + "_num"
 65            + self.add_leading_zeros(particulator.n_steps)
 66        )
 67        self.exported_times["attributes"][path] = particulator.n_steps * particulator.dt
 68        if self.verbose:
 69            print("Exporting Attributes to vtk, path: " + path)
 70        payload = {}
 71
 72        for k in particulator.attributes.keys():
 73            if len(particulator.attributes[k].shape) != 1:
 74                tmp = particulator.attributes[k].to_ndarray(raw=True)
 75                tmp_dict = {
 76                    k + "[" + str(i) + "]": tmp[i]
 77                    for i in range(len(particulator.attributes[k].shape))
 78                }
 79
 80                payload.update(tmp_dict)
 81            else:
 82                payload[k] = particulator.attributes[k].to_ndarray(raw=True)
 83
 84        payload.update(
 85            {
 86                k: np.array(v)
 87                for k, v in payload.items()
 88                if not (v.flags["C_CONTIGUOUS"] or v.flags["F_CONTIGUOUS"])
 89            }
 90        )
 91
 92        if particulator.mesh.dimension == 2:
 93            y = (
 94                particulator.mesh.size[0]
 95                / particulator.mesh.grid[0]
 96                * (payload["cell origin[0]"] + payload["position in cell[0]"])
 97            )
 98            x = (
 99                particulator.mesh.size[1]
100                / particulator.mesh.grid[1]
101                * (payload["cell origin[1]"] + payload["position in cell[1]"])
102            )
103            z = np.full_like(x, 0)
104        else:
105            raise NotImplementedError(
106                "Only 2 dimensions array is supported at the moment."
107            )
108
109        pointsToVTK(path, x, y, z, data=payload)
110
111    def export_products(self, particulator):
112        if len(particulator.products) != 0:
113            path = (
114                self.products_file_path
115                + "_num"
116                + self.add_leading_zeros(particulator.n_steps)
117            )
118            self.exported_times["products"][path] = (
119                particulator.n_steps * particulator.dt
120            )
121            if self.verbose:
122                print("Exporting Products to vtk, path: " + path)
123            payload = {}
124
125            if particulator.mesh.dimension != 2:
126                raise NotImplementedError(
127                    "Only 2 dimensions data is supported at the moment."
128                )
129
130            data_shape = (particulator.mesh.grid[1], particulator.mesh.grid[0], 1)
131
132            for k in particulator.products.keys():
133                v = particulator.products[k].get()
134
135                if isinstance(v, np.ndarray):
136                    if v.shape == particulator.mesh.grid:
137                        payload[k] = v[:, :, np.newaxis].copy()
138                    else:
139                        if self.verbose:
140                            print(
141                                f"{k} shape {v.shape} not equals data shape {data_shape}"
142                                f" and will not be exported",
143                                file=sys.stderr,
144                            )
145                elif isinstance(v, numbers.Number):
146                    if self.verbose:
147                        print(
148                            f"{k} is a Number and will not be exported", file=sys.stderr
149                        )
150                else:
151                    if self.verbose:
152                        print(f"{k} export is not possible", file=sys.stderr)
153
154            y, x, z = np.mgrid[
155                : particulator.mesh.grid[0] + 1, : particulator.mesh.grid[1] + 1, :1
156            ]
157            y = y * particulator.mesh.size[0] / particulator.mesh.grid[0]
158            x = x * particulator.mesh.size[1] / particulator.mesh.grid[1]
159            z = z * 1.0
160
161            gridToVTK(path, x, y, z, cellData=payload)
162        else:
163            if self.verbose:
164                print("No products to export")
165
166    def add_leading_zeros(self, a):
167        return "".join(["0" for i in range(self.num_len - len(str(a)))]) + str(a)

 45    def run(self, steps):
 46        for _ in range(steps):
 47            for key, dynamic in self.dynamics.items():
 48                with self.timers[key]:
 49                    dynamic()
 50            self.n_steps += 1
 51            self._notify_observers()

 53    def _notify_observers(self):
 54        reversed_order_so_that_environment_is_last = reversed(self.observers)
 55        for observer in reversed_order_so_that_environment_is_last:
 56            observer.notify()

107    def condensation(self, *, rtol_x, rtol_thd, counters, RH_max, success, cell_order):
108        """Updates droplet volumes by simulating condensation driven by prior changes
109          in environment thermodynamic state, updates the environment state.
110        In the case of parcel environment, condensation is driven solely by changes in
111          the dry-air density (theta and water_vapour_mixing_ratio should not be changed
112          by other dynamics).
113        In the case of prescribed-flow/kinematic environments, the dry-air density is
114          constant in time throughout the simulation.
115        This function should only change environment's predicted `thd` and
116          `water_vapour_mixing_ratio` (and not `rhod`).
117        """
118        self.backend.condensation(
119            solver=self.condensation_solver,
120            n_cell=self.mesh.n_cell,
121            cell_start_arg=self.attributes.cell_start,
122            water_mass=self.attributes["water mass"],
123            multiplicity=self.attributes["multiplicity"],
124            vdry=self.attributes["dry volume"],
125            idx=self.attributes._ParticleAttributes__idx,
126            rhod=self.environment["rhod"],
127            thd=self.environment["thd"],
128            water_vapour_mixing_ratio=self.environment["water_vapour_mixing_ratio"],
129            dv=self.environment.dv,
130            prhod=self.environment.get_predicted("rhod"),
131            pthd=self.environment.get_predicted("thd"),
132            predicted_water_vapour_mixing_ratio=self.environment.get_predicted(
133                "water_vapour_mixing_ratio"
134            ),
135            kappa=self.attributes["kappa"],
136            f_org=self.attributes["dry volume organic fraction"],
137            rtol_x=rtol_x,
138            rtol_thd=rtol_thd,
139            v_cr=self.attributes["critical volume"],
140            timestep=self.dt,
141            counters=counters,
142            cell_order=cell_order,
143            RH_max=RH_max,
144            success=success,
145            cell_id=self.attributes["cell id"],
146            reynolds_number=self.attributes["Reynolds number"],
147            air_density=self.environment["air density"],
148            air_dynamic_viscosity=self.environment["air dynamic viscosity"],
149        )
150        self.attributes.mark_updated("water mass")

310    def moments(
311        self,
312        *,
313        moment_0,
314        moments,
315        specs: dict,
316        attr_name="water mass",
317        attr_range=(-np.inf, np.inf),
318        weighting_attribute="water mass",
319        weighting_rank=0,
320        skip_division_by_m0=False,
321    ):
322        """
323        Writes to `moment_0` and `moment` the zero-th and the k-th statistical moments
324        of particle attributes computed filtering by value of the attribute `attr_name`
325        to fall within `attr_range`. The moment ranks are defined by `specs`.
326
327        Parameters:
328            specs: e.g., `specs={'volume': (1,2,3), 'kappa': (1)}` computes three moments
329                of volume and one moment of kappa
330            skip_division_by_m0: if set to `True`, the values written to `moments` are
331                multiplied by the 0-th moment (e.g., total volume instead of mean volume)
332        """
333        if len(specs) == 0:
334            raise ValueError("empty specs passed")
335        attr_data, ranks = [], []
336        for attr in specs:
337            for rank in specs[attr]:
338                attr_data.append(self.attributes[attr])
339                ranks.append(rank)
340        assert len(set(attr_data)) <= 1
341        if len(attr_data) == 0:
342            attr_data = self.backend.Storage.empty((0,), dtype=float)
343        else:
344            attr_data = attr_data[0]
345
346        ranks = self.backend.Storage.from_ndarray(np.array(ranks, dtype=float))
347
348        self.backend.moments(
349            moment_0=moment_0,
350            moments=moments,
351            multiplicity=self.attributes["multiplicity"],
352            attr_data=attr_data,
353            cell_id=self.attributes["cell id"],
354            idx=self.attributes._ParticleAttributes__idx,
355            length=self.attributes.super_droplet_count,
356            ranks=ranks,
357            min_x=attr_range[0],
358            max_x=attr_range[1],
359            x_attr=self.attributes[attr_name],
360            weighting_attribute=self.attributes[weighting_attribute],
361            weighting_rank=weighting_rank,
362            skip_division_by_m0=skip_division_by_m0,
363        )

442    def seeding(
443        self,
444        *,
445        seeded_particle_index,
446        seeded_particle_multiplicity,
447        seeded_particle_extensive_attributes,
448        number_of_super_particles_to_inject,
449    ):
450        n_null = self.n_sd - self.attributes.super_droplet_count
451        if n_null == 0:
452            raise ValueError(
453                "No available seeds to inject. Please provide particles with nan filled attributes."
454            )
455
456        if number_of_super_particles_to_inject > n_null:
457            raise ValueError(
458                "Trying to inject more super particles than space available."
459            )
460
461        if number_of_super_particles_to_inject > len(seeded_particle_multiplicity):
462            raise ValueError(
463                "Trying to inject multiple super particles with the same attributes. \
464                Instead increase multiplicity of injected particles."
465            )
466
467        self.backend.seeding(
468            idx=self.attributes._ParticleAttributes__idx,
469            multiplicity=self.attributes["multiplicity"],
470            extensive_attributes=self.attributes.get_extensive_attribute_storage(),
471            seeded_particle_index=seeded_particle_index,
472            seeded_particle_multiplicity=seeded_particle_multiplicity,
473            seeded_particle_extensive_attributes=seeded_particle_extensive_attributes,
474            number_of_super_particles_to_inject=number_of_super_particles_to_inject,
475        )
476        self.attributes.reset_idx()
477        self.attributes.sanitize()
478
479        self.attributes.mark_updated("multiplicity")
480        for key in self.attributes.get_extensive_attribute_keys():
481            self.attributes.mark_updated(key)

312        self,
313        *,
314        moment_0,
315        moments,
316        specs: dict,
317        attr_name="water mass",
318        attr_range=(-np.inf, np.inf),
319        weighting_attribute="water mass",
320        weighting_rank=0,
321        skip_division_by_m0=False,
322    ):
323        """
324        Writes to `moment_0` and `moment` the zero-th and the k-th statistical moments
325        of particle attributes computed filtering by value of the attribute `attr_name`
326        to fall within `attr_range`. The moment ranks are defined by `specs`.
327
328        Parameters:
329            specs: e.g., `specs={'volume': (1,2,3), 'kappa': (1)}` computes three moments
330                of volume and one moment of kappa
331            skip_division_by_m0: if set to `True`, the values written to `moments` are
332                multiplied by the 0-th moment (e.g., total volume instead of mean volume)
333        """
334        if len(specs) == 0:
335            raise ValueError("empty specs passed")
336        attr_data, ranks = [], []
337        for attr in specs:
338            for rank in specs[attr]:
339                attr_data.append(self.attributes[attr])
340                ranks.append(rank)
341        assert len(set(attr_data)) <= 1
342        if len(attr_data) == 0:
343            attr_data = self.backend.Storage.empty((0,), dtype=float)
344        else:
345            attr_data = attr_data[0]
346
347        ranks = self.backend.Storage.from_ndarray(np.array(ranks, dtype=float))
348
349        self.backend.moments(
350            moment_0=moment_0,
351            moments=moments,
352            multiplicity=self.attributes["multiplicity"],
353            attr_data=attr_data,
354            cell_id=self.attributes["cell id"],
355            idx=self.attributes._ParticleAttributes__idx,
356            length=self.attributes.super_droplet_count,
357            ranks=ranks,
358            min_x=attr_range[0],
359            max_x=attr_range[1],
360            x_attr=self.attributes[attr_name],
361            weighting_attribute=self.attributes[weighting_attribute],
362            weighting_rank=weighting_rank,
363            skip_division_by_m0=skip_division_by_m0,
364        )

443    def seeding(

447        seeded_particle_multiplicity,
448        seeded_particle_extensive_attributes,
449        number_of_super_particles_to_inject,
450    ):

476        )

478        self.attributes.sanitize()
